git checkout SulfurCamp // 내 branch를 들어갑니다. main에서 풀하거나 푸쉬를 하면 머리가 아파집니다.

git checkout main // main branch로 들어갑니다.

git add . // 모든 파일이 스테이징 상태로 갑니다. 올리기 1단계

git commit -m " " // " " 안에 commit 내용을 채우세요. 이번 주 풀어본 알고리즘, 코멘트를 다세요. 올리기 2단계

git push origin 본인 브랜치 이름 // 나는 git push origin SulfurCamp 올리기 3단계

git push origin main // 메인 브랜치에 있는 이번주 풀 문제, 공지 등 자동으로 로컬로 다운로드 됩니다.
